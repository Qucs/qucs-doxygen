01424 void Diagram::createSmithChart(Axis *Axis, int Mode)
01425 {
01426   int GridX;    // number of arcs with re(z)=const
01427   int GridY;    // number of arcs with im(z)=const
01428   calcSmithAxisScale(Axis, GridX, GridY);
01429 
01430 
01431   if(!xAxis.GridOn)  return;
01432 
01433   bool Zplane = ((Mode & 1) == 1);   // impedance or admittance chart ?
01434   bool Above  = ((Mode & 2) == 2);   // paint upper half ?
01435   bool Below  = ((Mode & 4) == 4);   // paint lower half ?
01436 
01437   int dx2 = x2>>1;
01438 
01439   double im, n_cos, n_sin, real, real1, real2, root;
01440   double rMAXq = Axis->up*Axis->up;
01441   int    theta, beta, phi, len, m, x, y;
01442 
01443   int R1 = int(x2/Axis->up + 0.5);
01444   // ....................................................
01445   // draw arcs with im(z)=const
01446   for(m=1; m<GridY; m++) {
01447     n_sin = pi*double(m)/double(GridY);
01448     n_cos = cos(n_sin);
01449     n_sin = sin(n_sin);
01450     im = (1.0-n_cos)/n_sin * pow(Axis->up,0.7); // up^0.7 is beauty correction
01451     y  = int(im/Axis->up*x2 + 0.5);  // diameter
01452 
01453     if(Axis->up <= 1.0) {       // Smith chart with |r|=1
01454       beta  = int(16.0*180.0*atan2(n_sin-im,n_cos-1.0)/pi - 0.5);
01455       if(beta<0) beta += 16*360;
01456       theta = 16*270-beta;
01457     }
01458     else {         // Smith chart with |r|>1
01459       im = 1.0/im;
01460       real = (rMAXq+1.0)/(rMAXq-1.0);
01461       root =  real*real - im*im - 1.0;
01462       if(root < 0.0) {  // circle lies completely within the Smith chart ?
01463         beta = 0;       // yes, ...
01464         theta = 16*360; // ... draw whole circle
01465       }
01466       else {
01467   // calculate both intersections with most outer circle
01468   real1 =  sqrt(root)-real;
01469   real2 = -sqrt(root)-real;
01470 
01471   root  = (real1+1.0)*(real1+1.0) + im*im;
01472   n_cos = (real1*real1 + im*im - 1.0) / root;
01473   n_sin = 2.0*im / root;
01474   beta  = int(16.0*180.0*atan2(n_sin-1.0/im,n_cos-1.0)/pi);
01475   if(beta<0) beta += 16*360;
01476 
01477   root  = (real2+1.0)*(real2+1.0) + im*im;
01478   n_cos = (real2*real2 + im*im - 1.0) / root;
01479   n_sin = 2.0*im / root;
01480   theta  = int(16.0*180.0*atan2(n_sin-1/im,n_cos-1)/pi);
01481   if(theta<0) theta += 16*360;
01482   theta = theta - beta;   // arc length
01483   if(theta < 0) theta = 16*360+theta;
01484       }
01485     }
01486 
01487     if(Zplane)
01488       x = (x2 + R1 - y) >> 1;
01489     else {
01490       x = (x2 - R1 - y) >> 1;
01491       beta = 16*180 - beta - theta;  // mirror
01492       if(beta < 0) beta += 16*360;   // angle has to be > 0
01493     }
01494 
01495     if(Above)
01496       Arcs.append(new struct Arc(x, dx2+y, y, y, beta, theta, GridPen));
01497     if(Below)

01499   }
01500 
01501   // ....................................................
01502   // draw  arcs with Re(z)=const
01503   theta = 0;       // arc length
01504   beta  = 16*180;  // start angle
01505   if(Above)  { beta = 0;  theta = 16*180; }
01506   if(Below)  theta += 16*180;
01507 
01508   for(m=1; m<GridX; m++) {
01509     im = m*(Axis->up+1.0)/GridX - Axis->up;
01510     y  = int((1.0-im)/Axis->up*double(dx2) + 0.5);  // diameter
01511 
01512     if(Zplane)
01513       x = ((x2+R1)>>1) - y;
01514     else
01515       x = (x2-R1)>>1;
01516     if(fabs(fabs(im)-1.0) > 0.2)   // if too near to |r|=1, it looks ugly
01517       Arcs.append(new struct Arc(x, (x2+y)>>1, y, y, beta, theta, GridPen));
01518 
01519     if(Axis->up > 1.0) {  // draw arcs on the rigth-handed side ?
01520       im = 1.0-im;
01521       im = (rMAXq-1.0)/(im*(im/2.0+1.0)) - 1.0;
01522       if(Zplane)  x += y;
01523       else  x -= y;

01525         Arcs.append(new struct Arc(x, (x2+y)>>1, y, y, beta, theta, GridPen));
01526       else {
01527         phi = int(16.0*180.0/pi*acos(im));
01528         len = 16*180-phi;

01533       }
01534     }
01535   }
01536 
01537 
01538   // ....................................................
01539   if(Axis->up > 1.0) {  // draw circle with |r|=1 ?
01540     x = (x2-R1) >> 1;

01676 bool Diagram::calcAxisScale(Axis *Axis, double& GridNum, double& zD,
01677         double& zDstep, double& GridStep, double Dist)
01678 {
01679   bool back=false;
01680   double numGrids, Base, Expo, corr;
01681 if(Axis->autoScale) {
01682 
01683   if(fabs(Axis->max-Axis->min) < 1e-200) {
01684     if((Axis->max == 0.0) && (Axis->min == 0.0)) {
01685       Axis->up  =  1.0;
01686       Axis->low = -1.0;
01687     }
01688     else {   // if max = min, double difference
01689       Axis->up  = Axis->max + fabs(Axis->max);
01690       Axis->low = Axis->min - fabs(Axis->min);
01691     }
01692   }
01693   else if(Axis != &xAxis) {
01694     // keep a small bounding between graph and  diagram limit
01695     Axis->up  = Axis->max + 0.1*(Axis->max-Axis->min);
01696     Axis->low = Axis->min - 0.1*(Axis->max-Axis->min);
01697   }
01698   else {
01699     Axis->up  = Axis->max;   // normal case for x axis
01700     Axis->low = Axis->min;
01701   }
01702 
01703 
01704   numGrids = floor(Dist/60.0);   // minimal grid is 60 pixel
01705   if(numGrids < 1.0) Base = Axis->up-Axis->low;
01706   else Base = (Axis->up-Axis->low)/numGrids;
01707   Expo = floor(log10(Base));
01708   Base = Base/pow(10.0,Expo);        // separate first significant digit
01709   if(Base < 3.5) {     // use only 1, 2 and 5, which ever is best fitted
01710     if(Base < 1.5) Base = 1.0;
01711     else Base = 2.0;
01712   }
01713   else {
01714     if(Base < 7.5) Base = 5.0;
01715     else { Base = 1.0; Expo++; }
01716   }
01717   GridStep = Base * pow(10.0,Expo);   // grid distance in real coordinates
01718   corr = floor((Axis->up-Axis->low)/GridStep - numGrids);
01719   if(corr < 0.0) corr++;
01720   numGrids += corr;     // correct rounding faults
01721 
01722 
01723   // upper y boundery ...........................
01724   zD = fabs(fmod(Axis->up, GridStep));// expand grid to upper diagram edge ?
01725   GridNum = zD/GridStep;
01726   if((1.0-GridNum) < 1e-10) GridNum = 0.0;  // fix rounding errors
01727   if(Axis->up <= 0.0) {
01728     if(GridNum < 0.3) { Axis->up += zD;  zD = 0.0; }
01729   }
01730   else  if(GridNum > 0.7)  Axis->up += GridStep-zD;
01731         else if(GridNum < 0.1)
01732          if(GridNum*Dist >= 1.0)// more than 1 pixel above ?
01733      Axis->up += 0.3*GridStep;  // beauty correction
01734 
01735 
01736   // lower y boundery ...........................
01737   zD = fabs(fmod(Axis->low, GridStep));// expand grid to lower diagram edge ?
01738   GridNum = zD/GridStep;
01739   if((1.0-GridNum) < 1e-10) zD = GridNum = 0.0;  // fix rounding errors
01740   if(Axis->low <= 0.0) {
01741     if(GridNum > 0.7) { Axis->low -= GridStep-zD;  zD = 0.0; }
01742     else if(GridNum < 0.1)
01743      if(GridNum*Dist >= 1.0) { // more than 1 pixel above ?
01744        Axis->low -= 0.3*GridStep;   // beauty correction
01745        zD += 0.3*GridStep;
01746      }
01747   }
01748   else {
01749     if(GridNum > 0.3) {
01750       zD = GridStep-zD;
01751       if(GridNum > 0.9) {
01752   if((1.0-GridNum)*Dist >= 1.0) { // more than 1 pixel above ?
01753     Axis->low -= 0.3*GridStep;    // beauty correction
01754     zD += 0.3*GridStep;
01755   }
01756       }
01757     }
01758     else { Axis->low -= zD; zD = 0.0; }
01759   }
01760 
01761   GridNum = Axis->low + zD;
01762   zD /= (Axis->up-Axis->low)/Dist;
01763 }
01764 else {   // user defined limits

01774

01776     zDstep = Dist;
01777     GridStep = Axis->step = Axis->up-Axis->low;
01778   }
