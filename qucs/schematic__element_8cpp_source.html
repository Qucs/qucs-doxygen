01534 // For moving elements: If the moving element is connected to a not
01535 // moving element, insert two wires. If the connected element is already
01536 // a wire, use this wire. Otherwise create new wire.
01537 void Schematic::newMovingWires(Q3PtrList<Element> *p, Node *pn, int pos)
01538 {
01539     Element *pe;
01540 
01541     if(pn->State & 8)  // Were new wires already inserted ?
01542         return;
01543     pn->State |= 8;
01544 
01545     for (;;)
01546     {
01547         if(pn->State & 16)  // node was already worked on
01548             break;
01549 
01550         pe = pn->Connections.getFirst();
01551         if(pe == 0)  return;
01552 
01553         if(pn->Connections.count() > 1)
01554             break;
01555         if(pe->Type != isWire)  // is it connected to exactly one wire ?
01556             break;
01557 
01558         // .................................................
01559         long  mask = 1, invMask = 3;
01560         Wire *pw2=0, *pw = (Wire*)pe;
01561 
01562         Node *pn2 = pw->Port1;
01563         if(pn2 == pn) pn2 = pw->Port2;
01564 
01565         if(pn2->Connections.count() == 2) // two existing wires connected ?
01566             if((pn2->State & (8+4)) == 0)
01567             {
01568                 Element *pe2 = pn2->Connections.getFirst();
01569                 if(pe2 == pe) pe2 = pn2->Connections.getLast();
01570                 // connected wire connected to exactly one wire ?
01571                 if(pe2->Type == isWire)
01572                     pw2  = (Wire*)pe2;
01573             }
01574 
01575         // .................................................
01576         // reuse one wire
01577         p->insert(pos, pw);
01578         pw->Port1->Connections.removeRef(pw);   // remove connection 1
01579         pw->Port1->State |= 16+4;
01580         pw->Port2->Connections.removeRef(pw);   // remove connection 2
01581         pw->Port2->State |= 16+4;
01582         Wires->take(Wires->findRef(pw));
01583 
01584         if(pw->isHorizontal()) mask = 2;
01585 
01586         if(pw2 == 0)    // place new wire between component and old wire
01587         {
01588             pn = pn2;
01589             mask ^= 3;
01590             invMask = 0;
01591         }
01592 
01593         if(pw->Port1 != pn)
01594         {
01595             pw->Port1->State |= mask;
01596             pw->Port1 = (Node*)mask;
01597             pw->Port2->State |= invMask;
01598             pw->Port2 = (Node*)invMask;  // move port 2 completely
01599         }
01600         else
01601         {
01602             pw->Port1->State |= invMask;
01603             pw->Port1 = (Node*)invMask;
01604             pw->Port2->State |= mask;
01605             pw->Port2 = (Node*)mask;
01606         }
01607 
01608         invMask ^= 3;
01609         // .................................................
01610         // create new wire ?
01611         if(pw2 == 0)
01612         {
01613             if(pw->Port1 != (Node*)mask)
01614                 p->insert(pos,
01615                           new Wire(pw->x2, pw->y2, pw->x2, pw->y2, (Node*)mask, (Node*)invMask));
01616             else
01617                 p->insert(pos,
01618                           new Wire(pw->x1, pw->y1, pw->x1, pw->y1, (Node*)mask, (Node*)invMask));
01619             return;
01620         }
01621 
01622 
01623         // .................................................
01624         // reuse a second wire
01625         p->insert(pos, pw2);
01626         pw2->Port1->Connections.removeRef(pw2);   // remove connection 1
01627         pw2->Port1->State |= 16+4;
01628         pw2->Port2->Connections.removeRef(pw2);   // remove connection 2
01629         pw2->Port2->State |= 16+4;
01630         Wires->take(Wires->findRef(pw2));
01631 
01632         if(pw2->Port1 != pn2)
01633         {
01634             pw2->Port1 = (Node*)0;
01635             pw2->Port2->State |= mask;
01636             pw2->Port2 = (Node*)mask;
01637         }
01638         else
01639         {
01640             pw2->Port1->State |= mask;
01641             pw2->Port1 = (Node*)mask;
01642             pw2->Port2 = (Node*)0;
01643         }
01644         return;
01645     }
01646 
01647     // only x2 moving
01648     p->insert(pos, new Wire(pn->cx, pn->cy, pn->cx, pn->cy, (Node*)0, (Node*)1));
01649     // x1, x2, y2 moving
01650     p->insert(pos, new Wire(pn->cx, pn->cy, pn->cx, pn->cy, (Node*)1, (Node*)3));
01651 }

01664 
01665

01667     // Insert components before wires in order to prevent short-cut removal.
01668     for(pc = Components->first(); pc != 0; )

01670         {
01671             p->append(pc);
01672             count++;

01674             // delete all port connections
01675             foreach(Port *pp, pc->Ports)
01676             {
01677                 pp->Connection->Connections.removeRef((Element*)pc);

01680

01684         else pc = Components->next();
01685 
01686     // test all wires and wire labels ***********************
01687     for(pw = Wires->first(); pw != 0; )
01688     {
01689         if(pw->Label) if(pw->Label->isSelected)
01690                 p->append(pw->Label);

01780         }
01781         else ppa = Paintings->next();
