01752 // ---------------------------------------------------
01753 // Determines the node names and writes subcircuits into netlist file.
01754 int Schematic::prepareNetlist(QTextStream& stream, QStringList& Collect,
01755                               QTextEdit *ErrText)
01756 {
01757   if(showBias > 0) showBias = -1;  // do not show DC bias anymore
01758 
01759   isVerilog = false;
01760   isAnalog = true;
01761   bool isTruthTable = false;
01762   int allTypes = 0, NumPorts = 0;
01763 
01764   // Detect simulation domain (analog/digital) by looking at component types.
01765   for(Component *pc = DocComps.first(); pc != 0; pc = DocComps.next()) {
01766     if(pc->isActive == COMP_IS_OPEN) continue;
01767     if(pc->Model.at(0) == '.') {
01768       if(pc->Model == ".Digi") {
01769         if(allTypes & isDigitalComponent) {
01770           ErrText->insert(
01771              QObject::tr("ERROR: Only one digital simulation allowed."));
01772           return -10;
01773         }
01774         if(pc->Props.getFirst()->Value != "TimeList")
01775           isTruthTable = true;
01776         if(pc->Props.getLast()->Value != "VHDL")
01777           isVerilog = true;
01778         allTypes |= isDigitalComponent;
01779         isAnalog = false;
01780       }
01781       else allTypes |= isAnalogComponent;
01782       if((allTypes & isComponent) == isComponent) {
01783         ErrText->insert(
01784            QObject::tr("ERROR: Analog and digital simulations cannot be mixed."));
01785         return -10;
01786       }
01787     }
01788     else if(pc->Model == "DigiSource") NumPorts++;
01789   }
01790 
01791   if((allTypes & isAnalogComponent) == 0) {
01792     if(allTypes == 0) {
01793       // If no simulation exists, assume analog simulation. There may
01794       // be a simulation within a SPICE file. Otherwise Qucsator will
01795       // output an error.
01796       isAnalog = true;
01797       allTypes |= isAnalogComponent;
01798       NumPorts = -1;
01799     }
01800     else {
01801       if(NumPorts < 1 && isTruthTable) {
01802         ErrText->insert(
01803            QObject::tr("ERROR: Digital simulation needs at least one digital source."));
01804         return -10;
01805       }
01806       if(!isTruthTable) NumPorts = 0;
01807     }
01808   }
01809   else {
01810     NumPorts = -1;
01811     isAnalog = true;
01812   }
01813 
01814   // first line is documentation
01815   if(allTypes & isAnalogComponent)
01816     stream << "#";
01817   else if (isVerilog)
01818     stream << "//";
01819   else
01820     stream << "--";
01821   stream << " Qucs " << PACKAGE_VERSION << "  " << DocName << "\n";
01822 
01823   // set timescale property for verilog schematics
01824   if (isVerilog) {
01825     stream << "\n`timescale 1ps/100fs\n";
01826   }
01827 
01828   int countInit = 0;  // counts the nodesets to give them unique names
01829   if(!giveNodeNames(&stream, countInit, Collect, ErrText, NumPorts))
01830     return -10;
01831 
01832   if(allTypes & isAnalogComponent)
01833     return NumPorts;
01834 
01835   if (!isVerilog) {
01836     stream << VHDL_LIBRARIES;
01837     stream << "entity TestBench is\n"
01838      << "end entity;\n"
01839      << "use work.all;\n";
01840   }
01841   return NumPorts;
01842 }
