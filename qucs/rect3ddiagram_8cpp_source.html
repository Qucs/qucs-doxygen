00062 // Calculates the coefficients for 3D -> 2D transformation
00063 void Rect3DDiagram::calcCoefficients()
00064 {
00065   double rX = double(rotX) * pi/180.0;
00066   double rY = double(rotY) * pi/180.0;
00067   double rZ = double(rotZ) * pi/180.0;
00068   
00069   cxy = sin(rZ);  cxx = cos(rZ);
00070   cxz = sin(rY);  rY  = cos(rY);
00071   cyz = sin(rX);  czz = cos(rX);
00072   rX  = cyz*cxz;  rZ = czz*cxz;
00073   cyx = czz * cxy + rX * cxx;
00074   cyy = czz * cxx - rX * cxy;
00075   czx = cyz * cxy - rZ * cxx;
00076   czy = cyz * cxx + rZ * cxy;
00077   cxx *=  rY;  cxy *= -rY;  cyz *= -rY;  czz *=  rY;
00078 }

00099 // Determines the position of the coordinate cross, i.e. calculates
00100 // "scaleX", "scaleY", "xorig" and "yorig". Returns the corner with
00101 // the largest distance from the viewer.
00102 int Rect3DDiagram::calcCross(int *Xses, int *Yses)
00103 {
00104   double x3D, y3D, z3D, x2D[8], y2D[8], z2D;
00105   double XMIN_2D, XMAX_2D, YMIN_2D, YMAX_2D, ZMIN_2D;
00106 
00107   int z, Center = 0;  // used to save minimum z (is center for axis cross)
00108   scaleX = scaleY = 1.0;  // used in "calcX_2D" and "calcY_2D"
00109   XMIN_2D = YMIN_2D = XMAX_2D = YMAX_2D = ZMIN_2D = 0.0; // origin is zero
00110   for(z=0; z<8; z++) {  // check 2D coordinates of all 8 quadrat corners
00111     if(z & 1) x3D = 1.0;  else x3D = 0.0;
00112     if(z & 2) y3D = 1.0;  else y3D = 0.0;
00113     if(z & 4) z3D = 1.0;  else z3D = 0.0;
00114     x2D[z] = calcX_2D(x3D, y3D, z3D);
00115     y2D[z] = calcY_2D(x3D, y3D, z3D);
00116     z2D = calcZ_2D(x3D, y3D, z3D);
00117 
00118     if(x2D[z] < XMIN_2D)  XMIN_2D = x2D[z];
00119     if(x2D[z] > XMAX_2D)  XMAX_2D = x2D[z];
00120     if(y2D[z] < YMIN_2D)  YMIN_2D = y2D[z];
00121     if(y2D[z] > YMAX_2D)  YMAX_2D = y2D[z];
00122     if(z2D < ZMIN_2D) { ZMIN_2D = z2D; Center = z; }
00123   }
00124 
00125   scaleX = double(x2) / (XMAX_2D - XMIN_2D); // scaling 3D -> 2D transformation
00126   scaleY = double(y2) / (YMAX_2D - YMIN_2D);
00127   xorig  = -XMIN_2D * scaleX;   // position of origin
00128   yorig  = -YMIN_2D * scaleY;
00129 
00130   for(z=0; z<8; z++) {  // calculate 2D coordinates of all corners
00131     *(Xses+z) = int(x2D[z] * scaleX + 0.5 + xorig);
00132     *(Yses+z) = int(y2D[z] * scaleY + 0.5 + yorig);
00133   }
00134   return Center;
00135 }

00629 // --------------------------------------------------------------
00630 int Rect3DDiagram::calcAxis(Axis *Axis, int x, int y,
00631                             double xD, double phi, bool Right)
00632 {
00633   Q_UNUSED(Right);
00634 
00635   double GridStep, corr, yD, stepD, GridNum, Expo;
00636   double xstepD, ystepD;
00637 
00638   QString tmp;
00639   // get size of text using the screen-compatible metric
00640   QFontMetrics metrics(QucsSettings.font, 0);
00641   int maxWidth = 0;
00642   int count, gx, gy, w;
00643 
00644   if(phi > 0.0)  Expo = phi - pi/2.0;
00645   else  Expo = phi + pi/2.0;
00646   gx = int(5.4 * cos(Expo) + 0.5);  // short grid marker lines
00647   gy = int(5.4 * sin(Expo) + 0.5);
00648 
00649   
00650   if(Axis->log) {
00651 
00652     bool back = calcAxisLogScale(Axis, w, yD, stepD, corr, int(xD));
00653 
00654     double upD  = Axis->up;
00655     if(yD > 1.5*stepD)  yD = 10.0*stepD;  // always start at power of 10
00656     if(back) {
00657       upD  = Axis->low;
00658       phi += pi;
00659       xD   = 0.0;
00660     }
00661 
00662     int xLen, yLen;
00663     ystepD = corr * log10(yD / fabs(Axis->low));
00664     while(ystepD <= xD) {  // create all grid lines
00665       
00666       tmp = misc::StringNiceNum(yD);
00667       if(Axis->up < 0.0)  tmp = '-'+tmp;
00668       w = metrics.width(tmp);  // width of text
00669       if(maxWidth < w) maxWidth = w;
00670       
00671       xLen = int(ystepD * cos(phi) + 0.5) + x;
00672       yLen = int(ystepD * sin(phi) + 0.5) + y;
00673       if(Qt::DockRight)
00674   Texts.append(new Text(xLen+3+gx, yLen-6+gy, tmp));
00675       else
00676   Texts.append(new Text(xLen-w-2-gx, yLen-6-gy, tmp));
00677       
00678       // short grid marks
00679       Lines.append(new Line(xLen-gx, yLen-gy, xLen+gx, yLen+gy,
00680           QPen(Qt::black,0)));
00681       yD *= 10.0;
00682       ystepD += corr;
00683     }
00684     
00685   }
00686   else {  // not logarithmical
00687     calcAxisScale(Axis, GridNum, yD, stepD, GridStep, xD);
00688     count = int((xD - yD) / stepD) + 1;  // number of grids
00689     
00690     xstepD = stepD * cos(phi);
00691     ystepD = stepD * sin(phi);
00692     xD = yD * cos(phi) + 0.5 + double(x);
00693     yD = yD * sin(phi) + 0.5 + double(y);
00694     
00695     if(Axis->up == 0.0)  Expo = log10(fabs(Axis->up-Axis->low));
00696     else  Expo = log10(fabs(Axis->up));
00697     
00698     for(; count>0; count--) {
00699       x = int(xD);
00700       y = int(yD);
00701       if(fabs(GridNum) < 0.01*pow(10.0, Expo)) GridNum = 0.0; // make 0 really 0
00702       tmp = misc::StringNiceNum(GridNum);
00703       
00704       w = metrics.width(tmp);  // width of text
00705       if(maxWidth < w) maxWidth = w;
00706       if(Qt::DockRight)
00707   Texts.append(new Text(x+3+gx, y-6+gy, tmp)); // place text right
00708       else
00709   Texts.append(new Text(x-w-2-gx, y-6-gy, tmp)); // place left
00710       GridNum += GridStep;
00711       
00712       // short grid marks
00713       Lines.append(new Line(x-gx, y-gy, x+gx, y+gy, QPen(Qt::black,0)));
00714       xD += xstepD;
00715       yD += ystepD;
00716     }
00717   } // of "if(Axis->log) ... else ..."
00718   
00719   return maxWidth+5;
00720 }

00725 {
00726   Q_UNUSED(Right);
00727 
00728   DataX *pD;

00734   // get size of text using the screen-compatible metric
00735   QFontMetrics metrics(QucsSettings.font, 0);
00736 
00737   x = x2_ - x1_;
00738   y = y2_ - y1_;
00739   cos_phi = sqrt(double(x*x) + double(y*y));
00740   phi = atan2(double(y), double(x));
00741   
00742   valid = calcAxis(Axis, x1_, y1_, cos_phi, phi, Qt::DockRight); // axis numbering
00743   z = (int)cos_phi;
00744   cos_phi = cos(phi);
00745   sin_phi = sin(phi);

00747   if(fabs(phi-1e-5) > pi/2.0) {

00749     y1_ = y2_;  sin_phi *= -1;
00750   }
00751   x = x1_ + int(double(valid)*sin_phi);
00752   y = y1_ - int(double(valid)*cos_phi);
00753   if(Axis->Label.isEmpty()) {
00754     // write all labels ----------------------------------------
00755     foreach(Graph *pg, Graphs) {
00756       if(Axis != &zAxis) {
00757         if(!pg->cPointsY)  continue;
00758         if(valid < 0) {
00759           delete[] pg->cPointsY;
00760           pg->cPointsY = 0;
00761           continue;
00762         }
00763         pD = pg->cPointsX.at(Index);
00764         if(!pD) continue;
00765         s = pD->Var;
00766       }
00767       else {
00768         s = pg->Var;
00769         if(!pg->cPointsY)  s += INVALID_STR;
00770       }
00771       x += int(double(metrics.lineSpacing())*sin_phi);
00772       y -= int(double(metrics.lineSpacing())*cos_phi);
00773       w = metrics.width(s);
00774       Texts.append(new Text(x+int(double((z-w)>>1)*cos_phi),
00775                             y+int(double((z-w)>>1)*sin_phi),
00776                             s, pg->Color, 12.0, cos_phi, sin_phi));
00777     }
00778   }
00779   else {
00780     x += int(double(metrics.lineSpacing())*sin_phi);
00781     y -= int(double(metrics.lineSpacing())*cos_phi);
00782     w = metrics.width(Axis->Label);
00783     Texts.append(new Text(x+int(double((z-w)>>1)*cos_phi),
00784                           y+int(double((z-w)>>1)*sin_phi),
00785                           Axis->Label, Qt::black, 12.0, cos_phi, sin_phi));
00786   }

00790 int Rect3DDiagram::calcDiagram()
00791 {
00792   Lines.clear();
00793   Texts.clear();
00794   Arcs.clear();
00795 
00796   double GridStep, corr, zD, zDstep, GridNum;
00797   // get size of text using the screen-compatible metric
00798   QFontMetrics metrics(QucsSettings.font, 0);
00799 
00800   x3 = x2 + 7;
00801   int z, z2, o, w;
00802 
00803   char *zBuffer=0;   // hidden line algorithm
00804   tBound *Bounds=0;
00805 
00806 
00807   // =====  give "step" the right sign  ==================================
00808   xAxis.step = fabs(xAxis.step);
00809   if(xAxis.limit_min > xAxis.limit_max)
00810     xAxis.step *= -1.0;
00811 
00812   yAxis.step = fabs(yAxis.step);
00813   if(yAxis.limit_min > yAxis.limit_max)
00814     yAxis.step *= -1.0;
00815 
00816   zAxis.step = fabs(zAxis.step);
00817   if(zAxis.limit_min > zAxis.limit_max)
00818     zAxis.step *= -1.0;
00819 
00820 
00821   // =====  calculate Axis.up and Axis.low  ==============================
00822   if(xAxis.log) {
00823     if(xAxis.autoScale) {
00824       if(xAxis.max*xAxis.min <= 0.0)  goto Frame;  // invalid
00825     }
00826     else  if(xAxis.limit_min*xAxis.limit_max <= 0.0)  goto Frame;  // invalid
00827     calcAxisLogScale(&xAxis, z, zD, zDstep, corr, x2);
00828   }
00829   else  calcAxisScale(&xAxis, GridNum, zD, zDstep, GridStep, double(x2));
00830     
00831   if(yAxis.log) {
00832     if(yAxis.autoScale) {
00833       if(yAxis.max*yAxis.min <= 0.0)  goto Frame;  // invalid
00834     }
00835     else  if(yAxis.limit_min*yAxis.limit_max <= 0.0)  goto Frame;  // invalid
00836     calcAxisLogScale(&yAxis, z, zD, zDstep, corr, x2);
00837   }
00838   else  calcAxisScale(&yAxis, GridNum, zD, zDstep, GridStep, double(x2));
00839 
00840   if(zAxis.log) {
00841     if(zAxis.autoScale) {
00842       if(zAxis.max*zAxis.min <= 0.0)  goto Frame;  // invalid
00843     }
00844     else  if(zAxis.limit_min*zAxis.limit_max <= 0.0)  goto Frame;  // invalid
00845     calcAxisLogScale(&zAxis, z, zD, zDstep, corr, x2);
00846   }
00847   else  calcAxisScale(&zAxis, GridNum, zD, zDstep, GridStep, double(x2));
00848 
00849 
00850   // ===  calculate transformation coefficients from rotation angles ===
00851   calcCoefficients();
00852 
00853   // =====  check calculate position of axes in 2D rectangle ===========
00854   int X[8], Y[8];
00855   o = calcCross(X, Y);
00856   // "o" is now the index of the origin coordinates.
00857 
00858 
00859   // =====  paint coordinate cross  ====================================
00860   // xy area
00861   Lines.append(new Line(X[o^1], Y[o^1], X[o^3], Y[o^3], QPen(Qt::black,0)));
00862   Lines.append(new Line(X[o^2], Y[o^2], X[o^3], Y[o^3], QPen(Qt::black,0)));
00863 
00864   // yz area
00865   Lines.append(new Line(X[o^2], Y[o^2], X[o^6], Y[o^6], QPen(Qt::black,0)));
00866   Lines.append(new Line(X[o^4], Y[o^4], X[o^6], Y[o^6], QPen(Qt::black,0)));
00867 
00868   // xz area
00869   Lines.append(new Line(X[o^1], Y[o^1], X[o^5], Y[o^5], QPen(Qt::black,0)));
00870   Lines.append(new Line(X[o^4], Y[o^4], X[o^5], Y[o^5], QPen(Qt::black,0)));
00871 
00872 
00873   // =====  create axis  =============================================
00874   if(X[o^1] < X[o^2])  w = 2;   // use which z axis ?
00875   else  w = 1;
00876 
00877   z = o^2;
00878   if(z & 1)  z ^= 1;  // from where to where ?
00879   z2 = z^1;
00880   createAxis(&xAxis, w == 2, X[z], Y[z], X[z2], Y[z2]);
00881 
00882   z = o^1;
00883   if(z & 2)  z ^= 2;  // from where to where ?
00884   z2 = z^2;
00885   createAxis(&yAxis, w == 1, X[z], Y[z], X[z2], Y[z2]);
00886 
00887   z = o^w;
00888   if(z & 4)  z ^= 4;  // from where to where ?
00889   z2 = z^4;
00890   createAxis(&zAxis, true, X[z], Y[z], X[z2], Y[z2]);
00891 
00892 
00893   if(hideLines) {
00894     w = (x2+1) * (y2/8 + 1);
00895     // To store the pixel coordinates that are already used (hidden).
00896     // Use one bit per pixel.
00897     zBuffer = (char*)malloc(w);
00898     memset(zBuffer, 0, w);
00899 
00900     // To store the boundings of the current polygon.
00901     Bounds = (tBound*)malloc((x2+1) * sizeof(tBound));
00902   }
00903 
00904   // hide invisible parts of graphs
00905   removeHiddenLines(zBuffer, Bounds);
00906 
00907   if(hideLines) {
00908     // now hide invisible part of coordinate cross
00909     tPoint3D *MemTmp = Mem;
00910     Mem = (tPoint3D*)malloc( 10*sizeof(tPoint3D) );
00911     
00912     removeHiddenCross(X[o^1], Y[o^1], X[o], Y[o], zBuffer, Bounds); // x axis
00913     removeHiddenCross(X[o^2], Y[o^2], X[o], Y[o], zBuffer, Bounds); // y axis
00914     removeHiddenCross(X[o^4], Y[o^4], X[o], Y[o], zBuffer, Bounds); // z axis
00915   
00916     free(Mem);
00917     Mem = MemTmp;  // write back values
00918 
00919     free(Bounds);
00920     free(zBuffer);
00921   }
00922   else {
00923     Lines.append(new Line(X[o], Y[o], X[o^1], Y[o^1], QPen(Qt::black,0)));
00924     Lines.append(new Line(X[o], Y[o], X[o^2], Y[o^2], QPen(Qt::black,0)));
00925     Lines.append(new Line(X[o], Y[o], X[o^4], Y[o^4], QPen(Qt::black,0)));
00926   }
00927 
00928   pMem = Mem;
00929   return 3;
00930 
00931 
00932 Frame:   // jump here if error occurred (e.g. impossible log boundings)
00933   Lines.append(new Line(0,  y2, x2, y2, QPen(Qt::black,0)));
00934   Lines.append(new Line(x2, y2, x2,  0, QPen(Qt::black,0)));
00935   Lines.append(new Line(0,   0, x2,  0, QPen(Qt::black,0)));
00936   Lines.append(new Line(0,  y2,  0,  0, QPen(Qt::black,0)));
00937   return 0;
00938 }

00941 // g->Points must already be empty!!!
00942 void Rect3DDiagram::calcData(Graph *g)
00943 {
00944   if(!pMem)  return;
00945   if(!g->cPointsY) return;

00947   int tmp;
00948   int Size = ((2*(g->cPointsX.getFirst()->count) + 1) * g->countY) + 10;
00949   Size *= 2;  // memory for cross grid lines
00950 
00951   double *py;
00952   if(g->countY > 1)  py = g->cPointsX.at(1)->Points;
00953 
00954   float *p = (float*)malloc( Size*sizeof(float) );  // create memory for points
00955   float *p_end;
00956   g->ScrPoints = p_end = p;
00957   p_end += Size - 9;   // limit of buffer
00958 
00959 
00960   *(p++) = STROKEEND;
00961   float dx=0.0, dy=0.0, xtmp=0.0, ytmp=0.0;
00962   double Stroke=10.0, Space=10.0; // length of strokes and spaces in pixel
00963   switch(g->Style) {
00964     case 0:      // ***** solid line **********************************
00965       do {
00966 
00967         while(1) {
00968           if(pMem->done & 11)    // is grid point ?
00969             if(pMem->done & 4) { // is hidden
00970               if(pMem > Mem) {
00971                 if((pMem-1)->done & 12)
00972                   break;
00973               }
00974               else  break;
00975       }
00976 
00977           FIT_MEMORY_SIZE;  // need to enlarge memory block ?
00978           *(p++) = pMem->x;
00979           *(p++) = pMem->y;
00980           break;
00981         }
00982 
00983         FIT_MEMORY_SIZE;  // need to enlarge memory block ?
00984         if(pMem->done & 8)  *(p++) = BRANCHEND;  // new branch
00985 
00986         if(pMem->done & 4)   // point invisible ?
00987           if( *(p-1) >= 0 )  // line already interrupted ?
00988             *(p++) = STROKEEND;
00989 
00990       } while(((pMem++)->done & 256) == 0);
00991       *p = GRAPHEND;
00992       return;
00993 
00994     case GRAPHSTYLE_DASH:
00995       Stroke = 10.0; Space =  6.0;
00996       break;
00997     case GRAPHSTYLE_DOT:
00998       Stroke =  2.0; Space =  4.0;
00999       break;
01000     case GRAPHSTYLE_LONGDASH:
01001       Stroke = 24.0; Space =  8.0;
01002       break;
01003 
01004     default:  // symbol (e.g. star) at each point **********************
01005       do {
01006         while(1) {
01007           if(pMem->done & 11)    // is grid point ?
01008             if(pMem->done & 4) { // is hidden
01009               if(pMem > Mem) {
01010                 if((pMem-1)->done & 12)
01011                   break;
01012               }
01013               else  break;
01014       }
01015 
01016           *(p++) = pMem->x;
01017           *(p++) = pMem->y;
01018           break;
01019         }
01020 
01021         if(pMem->done & 8)
01022           *(p++) = BRANCHEND;  // new branch
01023       } while(((pMem++)->done & 512) == 0);
01024       *p = GRAPHEND;
01025       return;
01026   }
01027 
01028 
01029   // **********  dashed, dotted, ...  *******************************
01030   int Counter = 0;  // counter for number of points in queue
01031   double alpha, dist = -Stroke;
01032   int Flag = 1;    // current state: 1=stroke, 0=space
01033   do {
01034     while(1) {
01035       if(pMem->done & 11)    // is grid point ?
01036         if(pMem->done & 4) { // is hidden
01037           if(pMem > Mem) {
01038             if((pMem-1)->done & 12)
01039               break;
01040           }
01041           else break;
01042   }
01043 
01044       xtmp = pMem->x;
01045       ytmp = pMem->y;
01046       Counter++;
01047       break;
01048     }
01049 
01050  
01051     if(Counter > 1) {
01052       if(Counter == 2) {
01053   *(p++) = dx;    // if first points of branch -> paint first one
01054   *(p++) = dy;
01055       }
01056       dx = xtmp - dx;
01057       dy = ytmp - dy;
01058       dist += sqrt(double(dx*dx + dy*dy)); // distance between points
01059       if((Flag == 1) && (dist <= 0.0)) {
01060   FIT_MEMORY_SIZE;  // need to enlarge memory block ?
01061   *(p++) = xtmp;    // if stroke then save points
01062   *(p++) = ytmp;
01063       }
01064       else {
01065         alpha = atan2(double(dy), double(dx));   // slope for interpolation
01066         while(dist > 0) {   // stroke or space finished ?
01067     FIT_MEMORY_SIZE;  // need to enlarge memory block ?
01068 
01069     *(p++) = xtmp - float(dist*cos(alpha)); // linearly interpolate
01070     *(p++) = ytmp - float(dist*sin(alpha));
01071 
01072           if(Flag == 0) {
01073             dist -= Stroke;
01074             if(dist <= 0) {
01075         *(p++) = xtmp;  // don't forget point after ...
01076         *(p++) = ytmp;  // ... interpolated point
01077             }
01078           }
01079           else {
01080       dist -= Space;
01081       if(*(p-3) < 0)  p -= 2;
01082       else *(p++) = STROKEEND;

01085         }
01086       }
01087 
01088     } // of "if(Counter > 1)"
01089 
01090 
01091     dx = xtmp;
01092     dy = ytmp;
01093 
01094     if(pMem->done & 8) {

01096         p -= 3;  // no single point after "no stroke"
01097       else if(*(p-3) == BRANCHEND) {

01099           p -= 2;  // erase last hidden point
01100       }
01101       *(p++) = BRANCHEND;  // new branch
01102       Counter = 0;
01103       Flag = 1;
01104       dist = -Stroke;
01105     }
01106 
01107     if(pMem->done & 4)   // point invisible ?
01108       if( *(p-1) >= 0 )  // line already interrupted ?
01109         *(p++) = STROKEEND;
01110 
01111   } while(((pMem++)->done & 256) == 0);
01112 
01113   *p = GRAPHEND;
01114

01120 for(int zz=0; zz<z; zz+=2)
01121   qDebug("c: %d/%d", *(p+zz), *(p+zz+1));
