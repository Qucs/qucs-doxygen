00076     activeAction = 0;   // no action active
00077     return false;
00078   }
00079 
00080   Schematic *Doc = (Schematic*)DocumentTab->currentPage();
00081   do {
00082     if(Function) if((Doc->*Function)()) {
00083       Action->blockSignals(true);
00084       Action->setChecked(false);  // release toolbar button
00085       Action->blockSignals(false);
00086       Doc->viewport()->update();
00087       break;
00088     }
00089 
00090     if(activeAction) {
00091       activeAction->blockSignals(true); // do not call toggle slot
00092       activeAction->setChecked(false);       // set last toolbar button off
00093       activeAction->blockSignals(false);
00094     }
00095     activeAction = Action;
00096 
00097     MouseMoveAction = MouseMove;
00098     MousePressAction = MousePress;
00099     MouseReleaseAction = 0;
00100     MouseDoubleClickAction = 0;
00101 
00102   } while(false);   // to perform "break"
00103 
00104   Doc->viewport()->update();
00105   view->drawn = false;
00106   return true;

01129   view->MAy1 = pc->cy + pc->ty;
01130 
01131   int z, n=0;  // "n" is number of property on screen
01132   pp = pc->Props.first();

01134     if(!pp) {  // should never happen
01135       slotHideEdit();
01136       return;
01137     }
01138     if(pp->display) n++;   // is visible ?
01139     pp = pc->Props.next();
01140   }
01141 
01142   pp = 0;
01143   if(view->MAx3 > 0)  pp = pc->Props.at(view->MAx3-1); // current property
01144   else s = pc->Name;
