00057 // This function replaces the LaTeX tags for special characters
00058 // into its unicode value.
00059 void encode_String(const QString& Input, QString& Output)
00060 {
00061   int Begin = 0, End = 0;
00062   struct tSpecialChar *p;
00063 
00064   Output = "";
00065   while((Begin=Input.indexOf('\\', Begin)) >= 0) {
00066     Output += Input.mid(End, Begin - End);
00067     End = Begin++;
00068 
00069     p = SpecialChars;
00070     while(p->Unicode != 0)   // test all special characters
00071       if(Input.mid(Begin).startsWith(p->Mnemonic)) {
00072         Output += QChar(p->Unicode);
00073         End = Begin + strlen(p->Mnemonic);
00074         break;
00075       }
00076       else p++;
00077   }
00078   Output += Input.mid(End);
00079 }
