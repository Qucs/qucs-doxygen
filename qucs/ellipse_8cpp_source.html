00237 void Ellipse::MouseResizeMoving(int x, int y, Schematic *p)
00238 {
00239   paintScheme(p);  // erase old painting
00240   switch(State) {
00241     case 0: x2 = x-cx; y2 = y-cy; // lower right corner
00242       break;
00243     case 1: x2 -= x-cx; cx = x; y2 = y-cy; // lower left corner
00244       break;
00245     case 2: x2 = x-cx; y2 -= y-cy; cy = y; // upper right corner
00246       break;
00247     case 3: x2 -= x-cx; cx = x; y2 -= y-cy; cy = y; // upper left corner
00248       break;
00249   }
00250   if(x2 < 0) { State ^= 1; x2 *= -1; cx -= x2; }
00251   if(y2 < 0) { State ^= 2; y2 *= -1; cy -= y2; }
00252 
00253   paintScheme(p);  // paint new painting

00313 // Checks if the coordinates x/y point to the painting.
00314 bool Ellipse::getSelected(float fX, float fY, float w)
00315 {
00316   float fX2 = float(x2);
00317   float fY2 = float(y2);
00318   fX -= float(cx) + fX2/2.0;
00319   fY -= float(cy) + fY2/2.0;
00320 
00321   if(filled) {
00322     float a = 2.0 * fX / fX2;  a *= a;
00323     float b = 2.0 * fY / fY2;  b *= b;
00324 
00325     if(a+b > 1.0)
00326       return false;
00327   }
00328   else {
00329     float a1 = fX / (fX2/2.0 - w);  a1 *= a1;
00330     float a2 = fX / (fX2/2.0 + w);  a2 *= a2;
00331     float b1 = fY / (fY2/2.0 - w);  b1 *= b1;
00332     float b2 = fY / (fY2/2.0 + w);  b2 *= b2;
00333 
00334     if(a1+b1 < 1.0)  return false;
00335     if(a2+b2 > 1.0)  return false;
00336   }
00337 
00338   return true;
