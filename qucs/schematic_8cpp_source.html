00867 // ---------------------------------------------------
00868 void Schematic::paintGrid(ViewPainter *p, int cX, int cY, int Width, int Height)
00869 {
00870   if(!GridOn) return;
00871 
00872   p->Painter->setPen(QPen(Qt::black,0));
00873   int dx = -int(Scale*float(ViewX1)) - cX;
00874   int dy = -int(Scale*float(ViewY1)) - cY;
00875   p->Painter->drawLine(-3+dx, dy, 4+dx, dy); // small cross at origin
00876   p->Painter->drawLine( dx,-3+dy, dx, 4+dy);
00877 
00878 
00879   int x1  = int(float(cX)/Scale) + ViewX1;
00880   int y1  = int(float(cY)/Scale) + ViewY1;
00881 
00883   setOnGrid(x1, y1);
00884   if(x1<0) x1 -= GridX - 1;
00885   else x1 += GridX;
00886   x1 -= x1 % (GridX << 1);
00887 
00888   if(y1<0) y1 -= GridY - 1;
00889   else y1 += GridY;
00890   y1 -= y1 % (GridY << 1);
00891 
00892   float X, Y, Y0, DX, DY;
00893   X = float(x1)*Scale + p->DX;
00894   Y = Y0 = float(y1)*Scale + p->DY;
00895   x1 = X > 0.0 ? int(X + 0.5) : int(X - 0.5);
00896   y1 = Y > 0.0 ? int(Y + 0.5) : int(Y - 0.5);
00897 
00898 
00899   int xEnd = x1 + Width;
00900   int yEnd = y1 + Height;
00901   DX = float(GridX << 1) * Scale;   // every second grid a point
00902   DY = float(GridY << 1) * Scale;
00903   while(DX <= 8.0)  DX *= 1.5;  // if too narrow, every third grid a point
00904   while(DY <= 8.0)  DY *= 1.5;  // if too narrow, every third grid a point
00905 
00906   while(x1 < xEnd) {
00907     Y = Y0;
00908     y1 = Y > 0.0 ? int(Y + 0.5) : int(Y - 0.5);
00909     while(y1 < yEnd) {
00910       p->Painter->drawPoint(x1, y1);    // paint grid
00911       Y += DY;
00912       y1 = Y > 0.0 ? int(Y + 0.5) : int(Y - 0.5);
00913     }
00914     X += DX;
00915     x1 = X > 0.0 ? int(X + 0.5) : int(X - 0.5);
00916   }
00917 }

01049     switch(pe->Type) {

01079       case isHWireLabel:
01080       case isVWireLabel:
01081   pl = (WireLabel*)pe;
01082   x2 = pl->x1;
