00905 /* This function runs the actual evaluation function and the returns
00906    the result. */
00907 constant * application::evaluate (void)
00908 {
00909     // Evaluate ddx() function.
00910     if (isDDX ())
00911     {
00912         if (getResult ()) delete getResult ();
00913         setResult (C (ddx->evaluate()->recreate ()));
00914         return getResult ();
00915     }
00916 
00917     int errors = 0;
00918     strlist * apreps = new strlist ();
00919 
00920     // first evaluate each argument
00921     for (node * arg = args; arg != NULL; arg = arg->getNext ())
00922     {
00923         // FIXME: Can save evaluation of already evaluated equations?
00924         if (arg->evaluated == 0 || 1)
00925         {
00926             arg->solvee = solvee;
00927             arg->evaluate ();

00930                 if (arg->getTag () == REFERENCE)
00931                 {
00932                     logprint (LOG_ERROR, "evaluate error, no such generated variable "
00933                               "`%s'\n", arg->toString ());
00934                 }
00935                 else
00936                 {
00937                     logprint (LOG_ERROR, "evaluate error, unable to evaluate "
00938                               "`%s'\n", arg->toString ());
00939                 }
00940                 errors++;
00941             }
00942             else
00943             {
00944                 // inherit drop/prep dependencies
00945                 if (arg->getResult()->dropdeps)
00946                 {
00947                     strlist * preps = arg->getResult()->getPrepDependencies ();
00948                     // recall longest prep dependencies' list of arguments
00949                     if (preps && (preps->length () > apreps->length ()))
00950                     {
00951                         delete apreps;
00952                         apreps = new strlist (*preps);
00953                     }
00954                 }
00955                 arg->evaluated++;
00956             }
00957         }
00958     }
00959 
00960     // then evaluate application itself
00961     if (!errors)
00962     {
00963         node * res;
00964         // delete previous result if necessary
00965         if ((res = getResult ()) != NULL) delete res;
00966         // then evaluate the application
00967         setResult (eval (C (args)));
00968         // check the returned type once again
00969         if (getResult()->getType () != getType ())
00970         {
00971             logprint (LOG_ERROR, "evaluate error, function `%s' returned invalid "
00972                       "constant type\n", toString ());
00973         }
00974     }
00975 
00976     // inherit prep dependencies of arguments if necessary
00977     if (!getResult()->dropdeps && apreps->length () > 0)
00978     {
00979         getResult()->dropdeps = 1;
00980         getResult()->appendPrepDependencies (apreps);
00981     }
00982     delete apreps;

01237 /* This function recursively finds the variable dependencies for each
01238    equation initially passed to the equation checker and returns the
01239    list of variable dependencies regarding this equation instance.
01240    The caller is responsible for deleting the returned string list
01241    object. */
01242 strlist * node::recurseDependencies (checker * check, strlist * deps)
01243 {
01244 
01245     strlist * res, * sub = NULL;
01246 
01247     /* The abort condition for recursion first. */
01248     if (deps->contains (A(this)->result))
01249     {
01250         res = new strlist (*deps);
01251         cycle = 1;
01252         return res;
01253     }
01254 
01255     /* Go through the list of passed dependency variables. */
01256     for (int i = 0; i < deps->length (); i++)
01257     {
01258         char * var = deps->get (i);
01259         node * child = check->findEquation (check->equations, var);
01260         /* Check each child equation. */
01261         if (child != NULL)
01262         {
01263             if (child->cycle == 0)
01264             {
01265                 strlist * cdeps = child->getDependencies ();
01266                 /* And append the dependencies of the child equation. */
01267                 if (cdeps->length () > 0)
01268                 {
01269                     res = strlist::join (sub, cdeps);
01270                     if (sub) delete sub;
01271                     sub = res;
01272                 }
01273             }
01274             /* If any child is cyclic, the parent is too. */
01275             else
01276             {
01277                 cycle = 1;
01278             }
01279         }
01280     }
01281 
01282     /* Recurse once again if the child equations revealed any more
01283        dependencies. */
01284     if (cycle && sub && sub->length () > 0)
01285     {
01286         res = recurseDependencies (check, sub);
01287         delete sub;
01288         sub = res;
01289     }
01290 
01291     /* Return the result. */
01292     res = strlist::join (deps, sub);
01293     if (sub) delete (sub);
01294     return res;
01295 }

01565 /* This function tries to find the given variable name which occurred
01566    in an equation dependency in the netlist.  If there is such a
01567    circuit property it returns a new assignment equation. */
01568 node * checker::findProperty (char * var)
01569 {
01570 
01571     node * eqn = NULL;
01572     int found = 0;
01573 
01574     // split into instance and property name
01575     char * ret, * inst, * prop;
01576     if ((ret = strchr (var, '.')) != NULL)
01577     {
01578         int len = ret - var;
01579         inst = (char *) calloc (1, len + 1);
01580         memcpy (inst, var, len);
01581         prop = &var[len + 1];
01582     }
01583     else return NULL;
01584 
01585     // go through list of circuit elements
01586     for (struct definition_t * def = defs; def; def = def->next)
01587     {
01588         if (!strcmp (def->instance, inst))
01589         {
01590             for (struct pair_t * pair = def->pairs; pair; pair = pair->next)
01591             {
01592                 if (!strcmp (pair->key, prop))
01593                 {
01594                     if (++found == 1)
01595                     {
01596                         if (pair->value->ident != NULL)
01597                         {
01598                             // reference
01599                             eqn = createReference ("#property", var, pair->value->ident);
01600                         }
01601                         else
01602                         {
01603                             // value
01604                             eqn = createDouble ("#property", var, pair->value->value);
01605                         }
01606                     }
01607                 }
01608             }
01609         }
01610     }
01611     if (found > 1)
01612     {
01613         logprint (LOG_ERROR, "checker error, desired property variable `%s' found "
01614                   "%dx, is not unique'\n", var, found);
01615         delete eqn;
01616         eqn = NULL;
01617     }
01618     else if (found == 1)
01619         appendEquation (eqn);
01620     free (inst);
01621     return eqn;
01622 }

01774 /* This function reorders the list of equations.  The new order can be
01775    used to evaluate the list step by step.  Each equation being
01776    evaluable is properly marked, remaining equations are appended. */
01777 void checker::reorderEquations (void)
01778 {
01779     node * root = NULL, * next, * last;
01780 
01781     // Go through the list of equations.
01782     for (node * eqn = equations; eqn != NULL; eqn = next)
01783     {
01784         strlist * deps = eqn->getDependencies ();
01785         int i, found, gens;
01786         next = eqn->getNext ();
01787         /* Check whether the variable dependencies can be found in
01788            previous equations. */
01789         for (found = gens = i = 0; i < deps->length (); i++)
01790         {
01791             char * var = deps->get (i);
01792             if (findEquation (root, var) != NULL) found++;
01793             if (isGenerated (var)) gens++;
01794         }
01795         // Yes.
01796         if (found == (deps->length () - gens))
01797         {
01798             /* Remove the equation from the current list and append it to
01799                the new list. */
01800             dropEquation (eqn);
01801             root = appendEquation (root, eqn);
01802             eqn->evalPossible = 1;
01803             // Now start over from the beginning.
01804             next = equations;
01805         }
01806     }
01807     // Any remaining equations get appended.
01808     if (root != NULL)
01809     {
01810         last = lastEquation (root);
01811         last->setNext (equations);
01812         equations = root;
01813     }
01814 }

02061 /* This function searches through the dataset of the equation solver
02062    for possible matrix vectors.  These are detected by the vectors'
02063    names (e.g. S[1,1]).  The matrix vectors found in the dataset get
02064    converted and saved into the set of equations.  */
02065 void solver::findMatrixVectors (qucs::vector * v)
02066 {
02067     qucs::vector * vec;
02068     strlist * deps;
02069     char * p, * cand;
02070     int s, r, c, a, b, n = 1;
02071 
02072     // initialize the  'found' flag
02073     for (vec = v; vec != NULL; vec = (qucs::vector *) vec->getNext ())
02074         vec->setRequested (0);
02075 
02076     // loop through the dataset vector until no more matrix vector is found
02077     do
02078     {
02079         r = c = s = -1;
02080         cand = NULL;
02081         deps = NULL;
02082         // go through the dataset
02083         for (vec = v; vec != NULL; vec = (qucs::vector *) vec->getNext ())
02084         {
02085             // skip detected vectors
02086             if (vec->getRequested ()) continue;
02087             // is the vector a possible matrix vector element ?
02088             if ((p = matvec::isMatrixVector (vec->getName (), a, b)) != NULL)
02089             {
02090                 if (cand != NULL)
02091                 {
02092                     // does this vectors name equals the current one ?
02093                     if (!strcmp (p, cand) && s == vec->getSize ())
02094                     {
02095                         // save largest row and column index and set the 'found' flag
02096                         if (a > r) r = a;
02097                         if (b > c) c = b;
02098                         vec->setRequested (n);
02099                     }
02100                 }
02101                 else
02102                 {
02103                     /* new possible matrix vector:
02104                        save its name, row and column index, its size (length of
02105                        the vector) and data dependencies; then set the 'found' flag */
02106                     cand = strdup (p);
02107                     r = a;
02108                     c = b;
02109                     s = vec->getSize ();
02110                     vec->setRequested (n);
02111                     deps = vec->getDependencies ();
02112                 }
02113                 free (p);
02114             }
02115         }
02116 
02117         // new matrix vector detected
02118         if (cand != NULL)
02119         {
02120             // create a new matrix vector and set the appropriate name
02121             matvec * mv = new matvec (s, r + 1, c + 1);
02122             mv->setName (cand);
02123             // go through the dataset vector once again
02124             for (vec = v; vec != NULL; vec = (qucs::vector *) vec->getNext ())
02125             {
02126                 // and collect the vectors with the same 'found' flags
02127                 if (vec->getRequested () == n)
02128                 {
02129                     p = matvec::isMatrixVector (vec->getName (), a, b);
02130                     mv->set (*vec, a, b);
02131                     free (p);
02132                     vec->setRequested (-1);
02133                 }
02134             }
02135             // now store this new matrix vector into the set of equations
02136             node * eqn = addEquationData (mv);
02137             eqn->solvee = this;
02138             eqn->evaluate ();
02139             if (deps == NULL)
02140             {
02141                 strlist * deps = new strlist ();
02142                 deps->add (mv->getName ());
02143                 eqn->setDataDependencies (deps);
02144                 delete deps;
02145             }
02146             else
02147             {
02148                 eqn->setDataDependencies (deps);
02149             }
02150             free (cand);
02151             cand = NULL;
02152         }
02153         // increase the current 'found' flag
02154         n++;
02155     }
02156     while (cand != NULL);
02157 }

02321 void solver::checkoutDataset (void)
02322 {
02323     // return if nothing todo
02324     if (data == NULL) return;

02326     foreach_equation (eqn)
02327     {
02328 
02329         // skip variables which don't need to be exported
02330         if (!eqn->output) continue;
02331 
02332         // is the equation result already in the dataset ?
02333         if (!findEquationResult (eqn))
02334         {
02335             qucs::vector * v = dataVector (eqn);
02336             if (v == NULL) continue;
02337 
02338             // collect inherited dataset dependencies
02339             strlist * datadeps = collectDataDependencies (eqn);
02340 
02341             // check whether dataset is smaller than its dependencies
02342             if (v->getSize () <= 1 && dataSize (datadeps) > v->getSize ())
02343             {
02344                 delete datadeps;
02345                 datadeps = NULL;
02346             }
02347 
02348             // store variable vector(s)
02349             if (datadeps && datadeps->length () > 0)
02350             {
02351                 v->setDependencies (datadeps);
02352                 if (v->getNext () != NULL)

02355                     data->addVariables (v);
02356                 }
02357                 else
02358                 {
02359                     data->addVariable (v);
02360                 }
02361             }
02362             // store independent vector(s)
02363             else
02364             {
02365                 if (v->getNext () != NULL)
02366                     data->addDependencies (v);
02367                 else
02368                     data->addDependency (v);

02370             }
02371         }
02372     }
