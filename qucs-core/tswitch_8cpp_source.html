00085 void tswitch::initTR (void) {
00086   qucs::vector * values = getPropertyVector ("time");
00087   // Find the total time of the switching periods
00088   T = real (sum (*values));
00089   // if the user enters an even number of switchng times
00090   // the pattern is repeated continuously
00091   repeat = (values->getSize () % 2) == 0 ? true : false;
00092   // make the time taken to go from fully on to fully off
00093   // the smallest switching time / 100, or the smallest possible
00094   // number, but no bogger than the max specified duration
00095   nr_double_t maxduration = getPropertyDouble("MaxDuration");
00096   duration = std::min ( std::max (10*NR_TINY, values->minimum() / 100),
00097                         maxduration );
00098 
00099   initDC ();
00100 }

00102 void tswitch::calcTR (nr_double_t t) {
00103   const char * const init = getPropertyString ("init");
00104   nr_double_t ron = getPropertyDouble ("Ron");
00105   nr_double_t roff = getPropertyDouble ("Roff");
00106   nr_double_t r = 0;
00107   nr_double_t rdiff = 0;
00108   nr_double_t s_i = 0;
00109   nr_double_t r_0 = 0;
00110   qucs::vector * values = getPropertyVector ("time");
00111   bool on = !strcmp (init, "on");
00112   nr_double_t ti = 0;
00113 
00114   if (repeat) {
00115     // if the user enters an even number of switchng times
00116     // the pattern is repeated continuously. This is acieved by
00117     // subtracting an integer number of total switching periods
00118     // from the real time
00119     t = t - T * qucs::floor (t / T);
00120   }
00121 
00122   // Initialise the last switching time to be a full
00123   // switching duration
00124   nr_double_t ts = t - duration;
00125 
00126   // here we determine whether a switching event should occur
00127   // by looping through the list of switching times and comparing
00128   // to the current time
00129   for (int i = 0; i < values->getSize (); i++) {
00130     // add the current value from the list of switching times
00131     // to a counter
00132     ti += real (values->get (i));
00133 
00134     if (t >= ti)
00135     {
00136       // the current time is greater than or equal to the current
00137       // sum of switching times so the switch state changes
00138       on = !on;
00139       // store the current switching time
00140       ts = ti;
00141     }
00142     else {
00143       // the current sum of switching times is in the future
00144       // so exit the loop
00145       break;
00146     }
00147   }
00148   // calculate the time since the last switch occured
00149   nr_double_t tdiff = std::max(NR_TINY, t - ts);
00150 
00151   // set the time difference to be no more than the max switch
00152   // duration so when we interpolate below we only get the max
00153   // or min function value if we are past a switching time
00154   if (tdiff > duration) {
00155         tdiff = duration;
00156   }
00157 
00158   // Set the appropriate resistance. The resistance is interpolated
00159   // along a cubic spline with zero derivative at the start and end
00160   // points to ensure a smooth derivative
00161   if (on)
00162   {
00163     r_0 = roff;
00164 
00165     rdiff = ron - roff;
00166 
00167     s_i = (rdiff) / (duration);
00168   }
00169   else
00170   {
00171     r_0 = ron;
00172 
00173     rdiff = roff - ron;
00174 
00175     s_i = (rdiff) / (duration);
00176   }
00177 
00178   // perform the interpolation of the constrained cubic spline
00179   r = r_0 + ((3. * s_i * qucs::pow (tdiff,2.0)) / (duration))
00180           + ((-2. * s_i * qucs::pow (tdiff,3.0)) / qucs::pow (duration, 2.0));
00181 
00182   setD (VSRC_1, VSRC_1, -r);
00183 }
