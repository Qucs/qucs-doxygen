00060 void iac::calcTR (nr_double_t t) {
00061   nr_double_t f = getPropertyDouble ("f");
00062   nr_double_t p = getPropertyDouble ("Phase");
00063   nr_double_t d = getPropertyDouble ("Theta");
00064   nr_double_t a = getPropertyDouble ("I");
00065   nr_double_t s = getNet()->getSrcFactor ();
00066   nr_double_t o = 2 * pi * f;
00067   nr_double_t T = p / f / 360;
00068   nr_double_t i = s * a * std::exp (-(t + T) * d * f) * std::sin (o * t + deg2rad (p));
00069   setI (NODE_1, +i); setI (NODE_2, -i);
00070 }
