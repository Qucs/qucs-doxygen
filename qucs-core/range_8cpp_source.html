00073 /* Checks whether the given value is inside the range. */
00074 bool range::inside (nr_double_t value) {
00075   int err = 0;
00076   if (il == '[' &&  (value < l))
00077     err++;
00078   if (il == ']' && !(value > l))
00079     err++;
00080   if (ih == '[' && !(value < h))
00081     err++;
00082   if (ih == ']' &&  (value > h))
00083     err++;
00084   return err == 0;
00085 }
