00050 void cpwline::ellipke (nr_double_t arg, nr_double_t &k, nr_double_t &e) {
00051   int iMax = 16;
00052   if (arg == 1.0) {
00053     k = std::numeric_limits<nr_double_t>::infinity();
00054     e = 0;
00055   }
00056   else if (std::isinf (arg) && arg < 0) {
00057     k = 0;
00058     e = std::numeric_limits<nr_double_t>::infinity();
00059   }
00060   else {
00061     nr_double_t a, b, c, f, s, fk = 1, fe = 1, t, da = arg;
00062     int i;
00063     if (arg < 0) {
00064       fk = 1 / qucs::sqrt (1 - arg);
00065       fe = qucs::sqrt (1 - arg);
00066       da = -arg / (1 - arg);
00067     }
00068     a = 1;
00069     b = qucs::sqrt (1 - da);
00070     c = qucs::sqrt (da);
00071     f = 0.5;
00072     s = f * c * c;
00073     for (i = 0; i < iMax; i++) {
00074       t = (a + b) / 2;
00075       c = (a - b) / 2;
00076       b = qucs::sqrt (a * b);
00077       a = t;
00078       f *= 2;
00079       s += f * c * c;
00080       if (c / a < std::numeric_limits<nr_double_t>::epsilon()) break;
00081     }
00082     if (i >= iMax) {
00083       k = 0; e = 0;
00084     }
00085     else {
00086       k = pi_over_2 / a;
00087       e = pi_over_2 * (1 - s) / a;
00088       if (arg < 0) {
00089         k *= fk;
00090         e *= fe;
00091       }
00092     }
00093   }
00094 }

00103 /* More or less accurate approximation of K(k)/K'(k).  Suggested by
00104    publications dealing with coplanar components. */
00105 nr_double_t cpwline::ellipa (nr_double_t k) {
00106   nr_double_t r, kp;
00107   if (k < sqrt1_2) {
00108     kp = qucs::sqrt (1 - k * k);
00109     r = pi / qucs::log (2 * (1 + qucs::sqrt (kp)) / (1 - qucs::sqrt (kp)));
00110   }
00111   else {
00112     r = qucs::log (2 * (1 + qucs::sqrt (k)) / (1 - qucs::sqrt (k))) / pi;
00113   }
00114   return r;
00115 }

00153   if (backMetal) {

00179     d  = (t * 1.25 / pi) * (1 + qucs::log (4 * pi * W / t));
00180     se = s - d;

00185     if (approx) {

00193       er0 = 1 + q3 * qz * (er - 1);

00201     // modifies er0 as well
00202     er0 = er0 - (0.7 * (er0 - 1) * t / s) / (q1 + (0.7 * t / s));

00209   // cut-off frequency of the TE0 mode
00210   fte = (C0 / 4) / (h * qucs::sqrt (er - 1));

00219   nr_double_t ac = 0;

00222     nr_double_t n  = (1 - k1) * 8 * pi / (t * (1 + k1));

00230 
00231   bt_factor  = 2 * pi / C0;

00234 void cpwline::calcAB (nr_double_t f, nr_double_t& zl, nr_double_t& al,
00235                       nr_double_t& bt) {
00236   nr_double_t sr_er_f = sr_er0;
00237   nr_double_t ac = ac_factor;
00238   nr_double_t ad = ad_factor;
00239 
00240   // by initializing as much as possible outside this function, the
00241   // overhead is minimal
00242 
00243   // add the dispersive effects to er0
00244   sr_er_f += (sr_er - sr_er0) / (1 + G * qucs::pow (f / fte, -1.8));
00245 
00246   // computes impedance
00247   zl /= sr_er_f;
00248 
00249   // for now, the loss are limited to strip losses (no radiation
00250   // losses yet) losses in neper/length
00251   ad *= f * (sr_er_f - 1 / sr_er_f);
00252   ac *= qucs::sqrt (f) * sr_er0;
00253 
00254   al  = ac + ad;
00255   bt *= sr_er_f * f;
00256 
00257   Er = sqr (sr_er_f);
00258   Zl = zl;
