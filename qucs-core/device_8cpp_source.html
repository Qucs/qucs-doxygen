00047 /* This function can be used to create an extra resistor circuit.  If
00048    the 'res' argument is NULL then the new circuit is created, the
00049    nodes get re-arranged and it is inserted into the given
00050    netlist. The given arguments can be explained as follows.
00051    base:     calling circuit (this)
00052    res:      additional resistor circuit (can be NULL)
00053    c:        name of the additional circuit
00054    n:        name of the inserted (internal) node
00055    internal: number of new internal node (the original external node) */
00056 circuit * device::splitResistor (circuit * base, circuit * res,
00057                                  const char * c, const char * n,
00058                                  int internal) {
00059   if (res == NULL) {
00060     res = new resistor ();
00061     const std::string &name = circuit::createInternal (c, base->getName ());
00062     const std::string &node = circuit::createInternal (n, base->getName ());
00063     res->setName (name);
00064     res->setNode (0, base->getNode(internal)->getName ());
00065     res->setNode (1, node, 1);
00066     base->getNet()->insertCircuit (res);
00067   }
00068   base->setNode (internal, res->getNode(1)->getName (), 1);
00069   return res;
00070 }

00116    order to avoid numerical overflows and thereby improve the
00117    convergence. */
00118 nr_double_t device::pnVoltage (nr_double_t Ud, nr_double_t Uold,
00119                                nr_double_t Ut, nr_double_t Ucrit) {
00120   nr_double_t arg;
00121   if (Ud > Ucrit && fabs (Ud - Uold) > 2 * Ut) {
00122     if (Uold > 0) {
00123       arg = (Ud - Uold) / Ut;
00124       if (arg > 0)
00125         Ud = Uold + Ut * (2 + log (arg - 2));
00126       else
00127         Ud = Uold - Ut * (2 + log (2 - arg));
00128     }
00129     else Ud = Uold < 0 ? Ut * log (Ud / Ut) : Ucrit;
00130   }
00131   else {
00132     if (Ud < 0) {
00133       arg = Uold > 0 ? -1 - Uold : 2 * Uold - 1;
00134       if (Ud < arg) Ud = arg;
00135     }
00136   }
00137   return Ud;

00259 nr_double_t
00260 device::fetVoltage (nr_double_t Ufet, nr_double_t Uold, nr_double_t Uth) {
00261   nr_double_t Utsthi = fabs (2 * (Uold - Uth)) + 2.0;
00262   nr_double_t Utstlo = Utsthi / 2;
00263   nr_double_t Utox   = Uth + 3.5;
00264   nr_double_t DeltaU = Ufet - Uold;
00265 
00266   if (Uold >= Uth) { /* FET is on */
00267     if (Uold >= Utox) {
00268       if (DeltaU <= 0) { /* going off */
00269         if (Ufet >= Utox) {
00270           if (-DeltaU > Utstlo) {
00271             Ufet = Uold - Utstlo;
00272           }
00273         } else {
00274           Ufet = MAX (Ufet, Uth + 2);
00275         }
00276       } else { /* staying on */
00277         if (DeltaU >= Utsthi) {
00278           Ufet = Uold + Utsthi;
00279         }
00280       }
00281     } else { /* middle region */
00282       if (DeltaU <= 0) { /* decreasing */
00283         Ufet = MAX (Ufet, Uth - 0.5);
00284       } else { /* increasing */
00285         Ufet = MIN (Ufet, Uth + 4);
00286       }
00287     }
00288   } else { /* FET is off */
00289     if (DeltaU <= 0) { /* staying off */
00290       if (-DeltaU > Utsthi) {
00291         Ufet = Uold - Utsthi;
00292       }
00293     } else { /* going on */
00294       if (Ufet <= Uth + 0.5) {
00295         if (DeltaU > Utstlo) {
00296           Ufet = Uold + Utstlo;
00297         }
00298       } else {
00299         Ufet = Uth + 0.5;
00300       }
00301     }

00327    the given voltages, surface potential and the zero-bias oxide
00328    capacitance. */
00329 void
00330 device::fetCapacitanceMeyer (nr_double_t Ugs, nr_double_t Ugd, nr_double_t Uth,
00331                              nr_double_t Udsat, nr_double_t Phi,
00332                              nr_double_t Cox, nr_double_t& Cgs,
00333                              nr_double_t& Cgd, nr_double_t& Cgb) {
00334 
00335   nr_double_t Utst = Ugs - Uth;
00336   if (Utst <= -Phi) { // cutoff regions
00337     Cgb = Cox;
00338     Cgs = 0;

00341     Cgb = -Utst * Cox / Phi;
00342     Cgs = 0;
00343     Cgd = 0;
00344   } else if (Utst <= 0) { // depletion region
00345     Cgb = -Utst * Cox / Phi;
00346     Cgs = Utst * Cox * 4 / 3 / Phi + 2 * Cox / 3;

00351     if (Udsat <= Uds) { // saturation region
00352       Cgs = 2 * Cox / 3;

00355       nr_double_t Sqr1 = sqr (Udsat - Uds);
00356       nr_double_t Sqr2 = sqr (2 * Udsat - Uds);
00357       Cgs = Cox * (1 - Sqr1 / Sqr2) * 2 / 3;
