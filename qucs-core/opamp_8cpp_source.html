00064 void opamp::calcDC (void) {
00065   nr_double_t g    = getPropertyDouble ("G");
00066   nr_double_t uMax = getPropertyDouble ("Umax");
00067   nr_double_t Uin  = real (getV (NODE_INP) - getV (NODE_INM));
00068   nr_double_t Uout = uMax * two_over_pi * qucs::atan (Uin * g * pi_over_2 / uMax);
00069   gv = g / (1 + sqr (pi_over_2 / uMax * g * Uin)) + GMin;
00070   setC (VSRC_1, NODE_INP, +gv);
00071   setC (VSRC_1, NODE_INM, -gv);
00072   setE (VSRC_1, Uin * gv - Uout);
00073 }
