00047 // Converts the given string into upper case.
00048 static char * spice_toupper (char * str) {
00049   for (unsigned int i = 0; i < strlen (str); i++) {
00050     if (str[i] >= 'a' && str[i] <= 'z') str[i] = toupper (str[i]);
00051   }
00052   return str;
00053 }
