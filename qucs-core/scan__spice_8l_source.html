00058 /* fixing invalid identifiers */
00059 static void spice_fix_identifier (char * ident) {
00060   char * p;
00061   for (p = ident; *p != '\0'; p++)
00062     switch (*p) {
00063     case '+': *p = 'P'; break;
00064     case '-': *p = 'N'; break;
00065     case '/': *p = '_'; break;
00066     }
00067 }
